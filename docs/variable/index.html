    public PackageJsonEditor: JsonEditor source 

    import {PackageJsonEditor} from 'tomo-cli/src/utils/index.js'

    Create and edit a package.json manifest file with a fluent API

    Example:

    Create a new package.json
    await (new PackageJsonEditor())
    .create()
    .commit();
    Create a new package.json and read its contents (chaining OK)
    const contents = (new PackageJsonEditor())
    .create()
    .read();
    Extend a package.json
    const script = {test: 'jest --coverage'};
await (new PackageJsonEditor())
    .extend({script})
    .commit();
    Create and extend a package.json without writing to disk (chaining OK)
    const script = {
    lint: 'eslint index.js -c ./.eslintrc.js'
};
await (new PackageJsonEditor())
    .create(false)
    .extend({script}, false)
    .commit();
