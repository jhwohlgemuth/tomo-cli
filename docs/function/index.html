    public async install(dependencies: string[], options: Object): string[] source 

    import {install} from 'tomo-cli/src/utils/index.js'

    Install dependencies with npm

    Params:

    NameTypeAttributeDescription
    dependencies string[]
    • optional
    • default: []

    Modules to install
    options Object

    Options to configure installation
    options.dev boolean
    • optional
    • default: false

    If true, add "--save-dev"
    options.latest boolean
    • optional
    • default: true

    If true, add "@latest" to all module names
    options.skipInstall boolean
    • optional
    • default: false

    Do not install (mostly for testing)

    Return:

    string[]

    Array of inputs (mostly for testing)

    Example:

    Install production dependencies
    install(['react']);
    Install development dependencies
    install(['jest', 'babel-jest'], {dev: true});
